// Example Processing Sketch
function setup() {
    createCanvas(800, 800);
    background(255);
    noLoop();
}

function draw() {
    for (let i = 0; i < 10; i++) {
        for (let j = 0; j < 10; j++) {
            fill(random(255), random(255), random(255));
            ellipse(i * 80 + 40, j * 80 + 40, 60, 60);
        }
    }
}
